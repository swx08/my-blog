@Service
public class ContentServiceImpl implements ContentService {

    @Autowired
    private RestHighLevelClient restHighLevelClient;

    /**
     * è§£ææ•°æ®å¹¶æ’å…¥åˆ°ESä¸­
     * @param keyword
     * @return
     * @throws IOException
     */
    @Override
    public boolean putContent(String keyword) throws IOException {
        //çˆ¬å–æ•°æ®
        List<Content> contentList = HtmlParseUtils.getContent(keyword);

        //å‘ESä¸­æ‰¹é‡æ’å…¥æ•°æ®
        BulkRequest bulkRequest = new BulkRequest();
        bulkRequest.timeout("2m");

        for (int i = 0; i < contentList.size(); i++) {
            bulkRequest.add(new IndexRequest(INDEX_NAME)
                    .source(JSON.toJSONString(contentList.get(i)), XContentType.JSON));
        }
        BulkResponse response = restHighLevelClient.bulk(bulkRequest, RequestOptions.DEFAULT);
        return !response.hasFailures();
    }

    /**
     * æ ¹æ®å…³é”®å­—åˆ†é¡µæŸ¥è¯¢ESä¸­çš„æ•°æ®
     * @param keyword
     * @param pageNo
     * @param pageSize
     * @return
     */
    @Override
    public List<Map<String, Object>> searchPage(String keyword, int pageNo, int pageSize) throws IOException {
        if(pageNo <= 1){
            pageNo = 1;
        }
        //æ¡ä»¶æœç´¢
        SearchRequest searchRequest = new SearchRequest(INDEX_NAME);
        SearchSourceBuilder sourceBuilder = new SearchSourceBuilder();
        //åˆ†é¡µ
        sourceBuilder.from(pageNo);
        sourceBuilder.size(pageSize);
        //æ„é€ æŸ¥è¯¢æ¡ä»¶
        TermQueryBuilder termQueryBuilder = QueryBuilders.termQuery("title", keyword);
        //æŸ¥è¯¢
        sourceBuilder.query(termQueryBuilder);
        //é«˜äº®æŸ¥è¯¢
        HighlightBuilder highlightBuilder = new HighlightBuilder();
        highlightBuilder.field("title");
        highlightBuilder.requireFieldMatch(false); //å…³é—­å¤šä¸ªé«˜äº®æ˜¾ç¤º
        highlightBuilder.preTags("<span style='color: red'>");
        highlightBuilder.postTags("</span>");
        sourceBuilder.highlighter(highlightBuilder);

        //æ‰§è¡ŒæŸ¥è¯¢è¯·æ±‚
        searchRequest.source(sourceBuilder);
        SearchResponse search = restHighLevelClient.search(searchRequest, RequestOptions.DEFAULT);
        //å°è£…æ•°æ®è¿”å›
        List<Map<String,Object>> searchList = new ArrayList<>();
        //è§£æç»“æœ
        for (SearchHit hit : search.getHits().getHits()) {
            //è§£æé«˜äº®å­—æ®µ,å°†åŸæ¥çš„å­—æ®µæ¢æˆé«˜äº®å­—æ®µå³å¯
            Map<String, HighlightField> highlightFields = hit.getHighlightFields();
            //è·å–é«˜äº®å­—æ®µ
            HighlightField highlightTitle = highlightFields.get("title");
            //åŸæ¥çš„ç»“æœ
            Map<String, Object> sourceAsMap = hit.getSourceAsMap();
            if(highlightTitle != null){
                Text[] fragments = highlightTitle.fragments();
                String new_title = "";
                for (Text text : fragments) {
                    new_title += text;
                }
                //å°†åŸæ¥çš„å­—æ®µæ¢æˆé«˜äº®å­—æ®µ
                sourceAsMap.put("title",new_title);
            }
            searchList.add(sourceAsMap);
        }
        return searchList;
    }
}
